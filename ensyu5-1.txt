演習問題5-1

・SQL文１

SELECT
	商品分類コード, 
	分類名
	count(商品分類コード) AS 商品数
FROM
	商品
  INNER JOIN
	商品分類
  ON
	商品.商品分類コード = 商品分類.商品分類コード
GROUP BY
	商品分類コード, 分類名
;

・SQL文２

SELECT
	社.支社名,
	店.支店名,
	商.商品名
FROM
(
	(
		(
			支社 AS 社
		INNER JOIN
			支店 AS 店
		ON
			社.支社コード = 店.支社コード
		)

	INNER JOIN
		支店商品 AS 支商
	ON
		店.支社コード = 支商.支社コード
	  AND
		店.支店コード = 支商.支店コード
	)
INNER JOIN
	商品 AS 商
ON
	支商.商品コード = 商.商品コード
)
;

・SQL文３

SELECT
	支店コード,
	MAX(count(商品数))
FROM
(
	(
		支店 AS 店
	INNER JOIN
		支店商品 AS 支商
	ON
		店.支社コード = 支商.支社コード
	  AND
		店.支店コード = 支商.支店コード
	)
INNER JOIN
	商品 AS 商
ON
	支商.商品コード = 商.商品コード
)
HAVING
	count(商品数) = MAX(count(商品数))
GROUP BY
	支店コード
;
